
 
 1. Objectiu de la Fase

L'objectiu d'aquesta fase és permetre que l'usuari, un cop s'ha autenticat correctament en el sistema, pugui visualitzar en una taula un llistat de tots els comptes bancaris dels quals és propietari o en té autorització, mostrant l'ID numèric intern de la base de dades, l'IBAN sencer i el seu estat actual (`ACTIU` o `BLOQUEJAT`).




2. Estructura del codi
Tot el codi es troba en un únic fitxer Program.cs: 

Main(): Bucle principal. Gestiona el login i el flux general de l'aplicació.

IntentarLogin(): Fa la consulta a la BBDD per validar l'usuari i la contrasenya (amb SHA2-256). Retorna true si es troba el registre.

MostrarMenuClient(): Mostra el menú de client i gestiona les opcions disponibles.

EjecutarCodigo1_0(): Consulta i mostra en format de taula tots els comptes vinculats a l'usuari autenticat.

LeerContrasenyaSegura(): Llegeix la contrasenya caràcter a caràcter i mostra asteriscs per pantalla en comptes del text real.

3: Explicacio


private static int usuariIdLoguejat = -1;
private static string usuariNomLoguejat = "";

Aquestes dues variables guarden l'id i el nom de l'usuari que ha iniciat sessió, per poder-los reutilitzar a totes les consultes posteriors sense haver de tornar a demanar-los.


Connexió a la base de dades:
private static string connectionString = 
    "SERVER=10.2.79.103;DATABASE=kaixa;UID=cashbox_app;PASSWORD=app123;";
On 10.2.79.103 es l'IP a la maquina virtual on estic gestionant el mysql.
S'utilitza l'usuari cashbox_app creat expressament per a l'aplicació, tal com indica l'enunciat.


Consulta de login:
SELECT id, nom FROM Usuaris 
WHERE username = @user AND password = SHA2(@pass, 256)
La contrasenya no es guarda en clar: es compara el hash SHA2-256 del que l'usuari introdueix amb el hash guardat a la BBDD. Si no coincideix no es retorna cap fila i el login falla.

Consulta de comptes:
SELECT c.id, c.numero_compte, c.estat 
FROM Comptes c
INNER JOIN UsuarisComptes uc ON c.id = uc.compte_id
WHERE uc.usuari_id = @idUsuari
La taula UsuarisComptes és la taula intermèdia que relaciona usuaris amb comptes 




4.Codi
using System;
using MySql.Data.MySqlClient;
using System.Collections.Generic;

class Program
{
    private static string connectionString = "SERVER=10.2.79.103;DATABASE=kaixa;UID=cashbox_app;PASSWORD=app123;";
    private static int usuariIdLoguejat = -1;
    private static string usuariNomLoguejat = "";

    static void Main(string[] args)
    {
        bool sortirAplicacio = false;

        while (!sortirAplicacio)
        {
            Console.Clear();
            Console.WriteLine("========================================");
            Console.WriteLine("         BENVINGUT A GESTKAIXA   ");
            Console.WriteLine("========================================");
            Console.Write("Usuari: ");
            string? username = Console.ReadLine(); 

            Console.Write("Contrasenya: ");
            string? password = LeerContrasenyaSegura(); 

            if (string.IsNullOrEmpty(username) || string.IsNullOrEmpty(password))
            {
                Console.WriteLine("\n[!] L'usuari i la contrasenya no poden estar buits.");
                Console.ReadKey();
                continue;
            }

            if (IntentarLogin(username, password))
            {
                MostrarMenuClient();
            }
            else
            {
                Console.WriteLine("\n[!] Error: Usuari o contrasenya incorrectes.");
                Console.Write("\nPrem 'S' per sortir o qualsevol altra tecla per reintentar... ");
                string? resposta = Console.ReadLine();
                if (resposta?.ToUpper() == "S")
                {
                    sortirAplicacio = true;
                }
            }
        }
    }

    static bool IntentarLogin(string username, string password)
    {
        string query = "SELECT id, nom FROM Usuaris WHERE username = @user AND password = SHA2(@pass, 256)";

        using (MySqlConnection conn = new MySqlConnection(connectionString))
        {
            try
            {
                conn.Open();
                using (MySqlCommand cmd = new MySqlCommand(query, conn))
                {
                    cmd.Parameters.AddWithValue("@user", username);
                    cmd.Parameters.AddWithValue("@pass", password);

                    using (MySqlDataReader reader = cmd.ExecuteReader())
                    {
                        if (reader.Read())
                        {
                            usuariIdLoguejat = Convert.ToInt32(reader["id"]);
                            usuariNomLoguejat = reader["nom"].ToString() ?? "Client";
                            return true; 
                        }
                    }
                }
            }
            catch (Exception ex)
            {
                Console.WriteLine($"\n[!] Error del servidor: {ex.Message}");
                Console.ReadKey();
            }
        }
        return false;
    }

    static void MostrarMenuClient()
    {
        bool tancarSessio = false;

        while (!tancarSessio)
        {
            Console.Clear();
            Console.WriteLine("========================================");
            Console.WriteLine($"=== MENÚ CLIENT | Usuari: {usuariNomLoguejat.ToUpper()} ===");
            Console.WriteLine("========================================");
            Console.WriteLine("1. Veure els meus comptes (Exercici 1.0)");
            Console.WriteLine("0. Tancar Sessió");
            Console.WriteLine("========================================");
            Console.Write("Selecciona una opció: ");

            string? opcio = Console.ReadLine();

            switch (opcio)
            {
                case "1":
                    EjecutarCodigo1_0(usuariIdLoguejat);
                    break;
                case "0":
                    tancarSessio = true;
                    usuariIdLoguejat = -1;
                    usuariNomLoguejat = "";
                    Console.WriteLine("\nSessió tancada.");
                    Console.ReadKey();
                    break;
                default:
                    Console.WriteLine("\nOpció no vàlida.");
                    Console.ReadKey();
                    break;
            }
        }
    }

    //1.0
    static void EjecutarCodigo1_0(int idUsuari)
    {
        Console.Clear();
        Console.WriteLine("========================================");
        Console.WriteLine("          ELS MEUS COMPTES        ");
        Console.WriteLine("========================================");

        string query = @"
            SELECT c.id, c.numero_compte, c.estat 
            FROM Comptes c
            INNER JOIN UsuarisComptes uc ON c.id = uc.compte_id
            WHERE uc.usuari_id = @idUsuari";

        using (MySqlConnection conn = new MySqlConnection(connectionString))
        {
            try
            {
                conn.Open();
                using (MySqlCommand cmd = new MySqlCommand(query, conn))
                {
                    cmd.Parameters.AddWithValue("@idUsuari", idUsuari);

                    using (MySqlDataReader reader = cmd.ExecuteReader())
                    {
                        if (!reader.HasRows)
                        {
                            Console.WriteLine("[!] No s'han trobat comptes vinculats.");
                        }
                        else
                        {
                            Console.WriteLine("{0,-10} {1,-25} {2,-15}", "ID", "Número de Compte", "Estat");
                            Console.WriteLine("------------------------------------------------");
                            while (reader.Read())
                            {
                                Console.WriteLine("{0,-10} {1,-25} {2,-15}", 
                                    reader["id"], 
                                    reader["numero_compte"], 
                                    reader["estat"]);
                            }
                        }
                    }
                }
            }
            catch (Exception ex)
            {
                Console.WriteLine($"[!] Error: {ex.Message}");
            }
        }
        Console.WriteLine("\nPrem una tecla per tornar al menú...");
        Console.ReadKey();
    }

    static string LeerContrasenyaSegura()
    {
        string pass = "";
        ConsoleKeyInfo key;
        do
        {
            key = Console.ReadKey(true);
            if (key.Key != ConsoleKey.Backspace && key.Key != ConsoleKey.Enter)
            {
                pass += key.KeyChar;
                Console.Write("*");
            }
            else if (key.Key == ConsoleKey.Backspace && pass.Length > 0)
            {
                pass = pass.Substring(0, (pass.Length - 1));
                Console.Write("\b \b");
            }
        } while (key.Key != ConsoleKey.Enter);
        Console.WriteLine();
        return pass;
      }
}
