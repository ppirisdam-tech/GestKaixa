
### 1. Objectiu de la Fase

Aquesta fase té com a finalitat oferir al client una visió ràpida de la seva situació financera. Ha de mostrar el saldo individualitzat de cadascun dels seus comptes i, al final, una línia destacada amb la suma total de tots els saldos, que representa el seu "Patrimoni Total".





2.Canvis


Respecte a la v1.0, els canvis principals són:

-El menú de client té 6 opcions completes: comptes, saldo, moviments, ingrés, retirada i alertes.
-S'ha afegit un segon tipus d'accés: l'administrador (cashbox_app / app123).
-L'administrador pot crear usuaris, obrir comptes i vincular-los.

AdminCrearCompte():Insereix un nou compte a Comptes

AdminAssignarUsuariACompte(): Insereix una relació a UsuarisComptes


4.Triggers

El trigger before_insert_moviments s'executa automàticament cada vegada que s'insereix un moviment, fa dos coses:

-Rebutja imports iguals a zero.
-Calcula el nou saldo acumulat sumant el saldo anterior i l'import nou

El trigger after_insert_moviments s'executa després de la inserció i comprova si el saldo ha creuat algo critic de llindar

-Si passa de positiu a negatiu insereix una alerta "Entrada en descobert".
-Si supera els 10.000 €  insereix una alerta "Supera llindar 10000".

using System;
using MySql.Data.MySqlClient;
using System.Collections.Generic;

class Program
{
    private static string connectionString = "SERVER=10.2.79.103;DATABASE=kaixa;UID=cashbox_app;PASSWORD=app123;";
    private static int usuariIdLoguejat = -1;
    private static string usuariNomLoguejat = "";

    static void Main(string[] args)
    {
        bool sortirAplicacio = false;

        while (!sortirAplicacio)
        {
            Console.Clear();
            Console.WriteLine("========================================");
            Console.WriteLine("         BENVINGUT A GESTKAIXA  ");
            Console.WriteLine("========================================");
            Console.Write("Usuari: ");
            string? username = Console.ReadLine(); 

            Console.Write("Contrasenya: ");
            string? password = LeerContrasenyaSegura(); 

            if (string.IsNullOrEmpty(username) || string.IsNullOrEmpty(password))
            {
                Console.WriteLine("\n[!] L'usuari i la contrasenya no poden estar buits.");
                Console.ReadKey();
                continue;
            }

            if (IntentarLogin(username, password))
            {
                MostrarMenuClient();
            }
            else
            {
                Console.WriteLine("\n[!] Error: Usuari o contrasenya incorrectes.");
                Console.Write("\nPrem 'S' per sortir o qualsevol altra tecla per reintentar... ");
                string? resposta = Console.ReadLine();
                if (resposta?.ToUpper() == "S")
                {
                    sortirAplicacio = true;
                }
            }
        }
    }

    static bool IntentarLogin(string username, string password)
    {
        string query = "SELECT id, nom FROM Usuaris WHERE username = @user AND password = SHA2(@pass, 256)";

        using (MySqlConnection conn = new MySqlConnection(connectionString))
        {
            try
            {
                conn.Open();
                using (MySqlCommand cmd = new MySqlCommand(query, conn))
                {
                    cmd.Parameters.AddWithValue("@user", username);
                    cmd.Parameters.AddWithValue("@pass", password);

                    using (MySqlDataReader reader = cmd.ExecuteReader())
                    {
                        if (reader.Read())
                        {
                            usuariIdLoguejat = Convert.ToInt32(reader["id"]);
                            usuariNomLoguejat = reader["nom"].ToString() ?? "Client";
                            return true; 
                        }
                    }
                }
            }
            catch (Exception ex)
            {
                Console.WriteLine($"\n[!] Error de connexió: {ex.Message}");
                Console.ReadKey();
            }
        }
        return false;
    }

    static void MostrarMenuClient()
    {
        bool tancarSessio = false;

        while (!tancarSessio)
        {
            Console.Clear();
            Console.WriteLine("========================================");
            Console.WriteLine($"=== MENÚ CLIENT | Usuari: {usuariNomLoguejat.ToUpper()} ===");
            Console.WriteLine("========================================");
            Console.WriteLine("1. Veure els meus comptes");
            Console.WriteLine("2. Consultar saldo total  ");
            Console.WriteLine("0. Tancar Sessió");
            Console.WriteLine("========================================");
            Console.Write("Selecciona una opció: ");

            string? opcio = Console.ReadLine();

            switch (opcio)
            {
                case "1":
                    EjecutarCodigo1_0(usuariIdLoguejat);
                    break;
                case "2":
                    EjecutarCodigo1_1(usuariIdLoguejat); 
                    break;
                case "0":
                    tancarSessio = true;
                    usuariIdLoguejat = -1;
                    usuariNomLoguejat = "";
                    Console.WriteLine("\nSessió tancada.");
                    Console.ReadKey();
                    break;
                default:
                    Console.WriteLine("\nOpció no vàlida.");
                    Console.ReadKey();
                    break;
            }
        }
    }

    static void EjecutarCodigo1_0(int idUsuari)
    {
        Console.Clear();
        Console.WriteLine("========================================");
        Console.WriteLine("          ELS MEUS COMPTES        ");
        Console.WriteLine("========================================");

        string query = @"
            SELECT c.id, c.numero_compte, c.estat 
            FROM Comptes c
            INNER JOIN UsuarisComptes uc ON c.id = uc.compte_id
            WHERE uc.usuari_id = @idUsuari";

        using (MySqlConnection conn = new MySqlConnection(connectionString))
        {
            try
            {
                conn.Open();
                using (MySqlCommand cmd = new MySqlCommand(query, conn))
                {
                    cmd.Parameters.AddWithValue("@idUsuari", idUsuari);

                    using (MySqlDataReader reader = cmd.ExecuteReader())
                    {
                        if (!reader.HasRows)
                        {
                            Console.WriteLine("[!] No s'han trobat comptes.");
                        }
                        else
                        {
                            Console.WriteLine("{0,-10} {1,-25} {2,-15}", "ID", "Número de Compte", "Estat");
                            Console.WriteLine("------------------------------------------------");
                            while (reader.Read())
                            {
                                Console.WriteLine("{0,-10} {1,-25} {2,-15}", 
                                    reader["id"], 
                                    reader["numero_compte"], 
                                    reader["estat"]);
                            }
                        }
                    }
                }
            }
            catch (Exception ex)
            {
                Console.WriteLine($"[!] Error: {ex.Message}");
            }
        }
        Console.WriteLine("\nPrem una tecla per tornar al menú...");
        Console.ReadKey();
    }

    //1.1
    static void EjecutarCodigo1_1(int idUsuari)
    {
        Console.Clear();
        Console.WriteLine("========================================");
        Console.WriteLine("       SALDO TOTAL DELS COMPTES   ");
        Console.WriteLine("========================================");

        string query = @"
            SELECT c.numero_compte, IFNULL(vs.saldo, 0) AS saldo
            FROM Comptes c
            INNER JOIN UsuarisComptes uc ON c.id = uc.compte_id
            LEFT JOIN VistaSaldos vs ON c.id = vs.compte_id
            WHERE uc.usuari_id = @idUsuari";

        decimal totalPatrimoni = 0;

        using (MySqlConnection conn = new MySqlConnection(connectionString))
        {
            try
            {
                conn.Open();
                using (MySqlCommand cmd = new MySqlCommand(query, conn))
                {
                    cmd.Parameters.AddWithValue("@idUsuari", idUsuari);

                    using (MySqlDataReader reader = cmd.ExecuteReader())
                    {
                        if (!reader.HasRows)
                        {
                            Console.WriteLine("[!] No es disposa de dades de saldos.");
                        }
                        else
                        {
                            Console.WriteLine("{0,-25} {1,15}", "Número de Compte", "Saldo");
                            Console.WriteLine("----------------------------------------");
                            
                            while (reader.Read())
                            {
                                decimal saldoCompte = Convert.ToDecimal(reader["saldo"]);
                                totalPatrimoni += saldoCompte;

                                Console.WriteLine("{0,-25} {1,14:C2}", reader["numero_compte"], saldoCompte);
                            }

                            Console.WriteLine("----------------------------------------");
                            Console.ForegroundColor = ConsoleColor.Green;
                            Console.WriteLine("{0,-25} {1,14:C2}", "TOTAL PATRIMONI:", totalPatrimoni);
                            Console.ResetColor();
                        }
                    }
                }
            }
            catch (Exception ex)
            {
                Console.WriteLine($"[!] Error en calcular els saldos: {ex.Message}");
            }
        }
        Console.WriteLine("\nPrem una tecla per tornar al menú...");
        Console.ReadKey();
    }

    static string LeerContrasenyaSegura()
    {
        string pass = "";
        ConsoleKeyInfo key;
        do
        {
            key = Console.ReadKey(true);
            if (key.Key != ConsoleKey.Backspace && key.Key != ConsoleKey.Enter)
            {
                pass += key.KeyChar;
                Console.Write("*");
            }
            else if (key.Key == ConsoleKey.Backspace && pass.Length > 0)
            {
                pass = pass.Substring(0, (pass.Length - 1));
                Console.Write("\b \b");
            }
        } while (key.Key != ConsoleKey.Enter);
        Console.WriteLine();
        return pass;
    }
}
